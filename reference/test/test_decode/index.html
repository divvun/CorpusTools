    def handler(err):
    """Handle UnicodeDecodeError.

    Args:
        err (exceptions.UnicodeDecodeError): the error.

    Returns:
        (str): The fixed string
    """
    start = err.start
    end = err.end
    return (
        "".join([f"&#{err.object[i]};" for i in range(start, end)]),
        end,
    )
